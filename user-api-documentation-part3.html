电费账单系统 - 用户端API文档

第三部分：帮助中心

帮助中心

帮助中心模块提供用户搜索帮助内容、获取分类问题列表、提交问题反馈等功能。

搜索帮助内容

GET
/api/user/help/search

搜索帮助中心内容

前端实现代码:
import request from '@/utils/request';

/**
 * 搜索帮助内容
 * @param {Object} params 查询参数
 * @param {string} params.query 搜索关键词
 * @returns {Promise} 返回Promise对象，包含搜索结果
 */
export function searchHelp(params) {
  return request({
    url: '/api/user/help/search',
    method: 'get',
    params
  });
}

参数：

参数名 类型 必填 说明
query string 是 搜索关键词

返回数据：

{
    "results": [                 // 搜索结果
        {
            "id": "string",      // 问题ID
            "question": "string", // 问题标题
            "answer": "string",  // 问题答案
            "category": "string", // 问题分类
            "links": [           // 相关链接
                {
                    "text": "string", // 链接文本
                    "url": "string"   // 链接地址
                }
            ]
        }
    ]
}

获取分类问题列表

GET
/api/user/help/category/{categoryId}

获取指定分类下的所有问题

前端实现代码:
/**
 * 获取分类问题列表
 * @param {string} categoryId 分类ID
 * @returns {Promise} 返回Promise对象，包含分类问题列表
 */
export function getCategoryFAQs(categoryId) {
  return request({
    url: `/api/user/help/category/${categoryId}`,
    method: 'get'
  });
}

参数：

参数名 类型 必填 说明
categoryId string 是 分类ID

返回数据：

{
    "category": {                // 分类信息
        "id": "string",          // 分类ID
        "name": "string",        // 分类名称
        "icon": "string"         // 分类图标
    },
    "faqs": [                    // 问题列表
        {
            "id": "string",      // 问题ID
            "question": "string", // 问题标题
            "answer": "string",  // 问题答案
            "links": [           // 相关链接
                {
                    "text": "string", // 链接文本
                    "url": "string"   // 链接地址
                }
            ]
        }
    ]
}

提交问题反馈

POST
/api/user/help/feedback

提交帮助内容反馈

前端实现代码:
/**
 * 提交问题反馈
 * @param {Object} data 反馈数据
 * @param {string} data.faqId 问题ID
 * @param {boolean} data.helpful 是否有帮助
 * @param {string} data.comment 反馈意见（可选）
 * @returns {Promise} 返回Promise对象，包含提交结果
 */
export function submitFeedback(data) {
  return request({
    url: '/api/user/help/feedback',
    method: 'post',
    data
  });
}

请求参数：

参数名 类型 必填 说明
faqId string 是 问题ID
helpful boolean 是 是否有帮助
comment string 否 反馈意见

提交问题反馈表单

export function submitFeedbackForm(data) {

请求参数：

参数名 类型 必填 说明
title string 是 问题标题
content string 是 问题描述
category string 是 问题分类
contact string 否 联系方式
attachments array 否 附件列表

获取热门问题

GET
/api/user/help/hot-questions

获取热门问题列表

前端实现代码:
/**
 * 获取热门问题
 * @param {Object} params 查询参数
 * @param {number} params.limit 获取数量限制（可选）
 * @returns {Promise} 返回Promise对象，包含热门问题列表
 */
export function getHotQuestions(params) {
  return request({
    url: '/api/user/help/hot-questions',
    method: 'get',
    params
  });
}

参数：

参数名 类型 必填 说明
limit number 否 返回问题的数量限制，默认10

返回数据：

{
    "questions": [               // 热门问题列表
        {
            "id": "string",      // 问题ID
            "question": "string", // 问题标题
            "answer": "string",  // 问题答案
            "category": "string", // 问题分类
            "viewCount": "number" // 浏览次数
        }
    ]
}

获取操作指南

GET
/api/user/help/guides

获取操作指南，可选择获取特定指南或所有指南

前端实现代码:
/**
 * 获取操作指南
 * @param {string} guideId 指南ID（可选）
 * @returns {Promise} 返回Promise对象，包含操作指南
 */
export function getOperationGuide(guideId) {
  return request({
    url: '/api/user/help/guides',
    method: 'get',
    params: guideId ? { guideId } : {}
  });
}

参数：

参数名 类型 必填 说明
guideId string 否 指南ID，不传则返回所有指南

返回数据：

// 单个指南时返回:
{
    "guide": {
        "id": "string",          // 指南ID
        "title": "string",       // 指南标题
        "content": "string",     // 指南内容（富文本）
        "category": "string",    // 指南分类
        "steps": [               // 操作步骤
            {
                "step": "number",// 步骤序号
                "title": "string", // 步骤标题
                "content": "string", // 步骤内容
                "imageUrl": "string" // 步骤图片
            }
        ],
        "videoUrl": "string",    // 教学视频地址（可选）
        "relatedGuides": [       // 相关指南
            {
                "id": "string",  // 相关指南ID
                "title": "string" // 相关指南标题
            }
        ]
    }
}

// 所有指南时返回:
{
    "guides": [                  // 指南列表
        {
            "id": "string",      // 指南ID
            "title": "string",   // 指南标题
            "category": "string", // 指南分类
            "description": "string", // 简要描述
            "thumbnailUrl": "string" // 缩略图地址
        }
    ],
    "categories": [              // 指南分类
        {
            "id": "string",      // 分类ID
            "name": "string",    // 分类名称
            "count": "number"    // 指南数量
        }
    ]
}
